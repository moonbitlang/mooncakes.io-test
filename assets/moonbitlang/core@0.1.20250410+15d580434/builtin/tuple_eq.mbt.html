///|
pub impl[T0 : Eq, T1 : Eq] Eq for (T0, T1) with op_equal(
  self : (T0, T1),
  other : (T0, T1)
) -> Bool {
  self.0 == other.0 && self.1 == other.1
}

///|

  self : (T0, T1, T2),
  other : (T0, T1, T2)

  self.0 == other.0 && self.1 == other.1 && self.2 == other.2
}
